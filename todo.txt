CREATE VIEW credits_view AS
select comics.id, persons.name as person, roles.name as role
from comics 
inner join credits
   on credits.comic_id=comics.id
inner join persons
   on credits.person_id=persons.id
inner join roles 
   on credits.role_id=roles.id
where comics.id=1
